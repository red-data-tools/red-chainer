def initialize(ksize, stride: nil, pad: 0, cover_all: true)
  if stride.nil?
    stride = ksize
  end

  @kh, @kw = ksize.is_a?(::Array) ? ksize : [ksize, ksize]
  @sy, @sx = stride.is_a?(::Array) ? stride : [stride, stride]
  @ph, @pw = pad.is_a?(::Array) ? pad: [pad, pad]

  @cover_all = cover_all
  @use_cudnn = false
end
