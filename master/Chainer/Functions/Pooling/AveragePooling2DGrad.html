def forward(gy)
  if @use_cudnn
    return _forward_cudnn(gy[0])
  end

  h, w  = @in_shape[2..-1]
  shape = gy[0].shape
  shape.insert(2, 1, 1)
  gcol = gy[0].reshape(*shape).tile(1, 1, @kh, @kw, 1, 1)

  gx = Chainer::Utils::Conv.col2im(gcol, @sy, @sx, @ph, @pw, h, w)
  gx /= @kh * @kw
  [gx]
end
