def self.logsumexp(x)
  xm = Chainer.get_array_module(x)
  m = x.max(axis: 1, keepdims: true)
  y = x - m
  y = xm::NMath.exp(y)
  s = y.sum(axis: 1, keepdims: true)
  s = xm::NMath.log(s)
  m + s
end
