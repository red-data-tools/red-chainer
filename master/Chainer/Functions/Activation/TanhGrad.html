def backward(indexes, grad_outputs)
  y, gy = get_retained_inputs
  g = grad_outputs[0]

  y_mul_g = y * g
  grad_y = -2 * gy * y_mul_g
  ggy = g - y * y_mul_g
  [grad_y, ggy]
end

def forward(inputs)
  retain_inputs([0, 1])
  y, gy = inputs

  one = y.class.new.fill(1)
  [Utils::Array.force_array(gy * (one - y * y))]
end
